1114. 按序打印 - 力扣（LeetCode）

解法一：synchronized + wait/notifyAll

class Foo {

    final Object lock = new Object();

    volatile int count = 1;

    public Foo() {
    }

    public void first(Runnable printFirst) throws InterruptedException {

        // printFirst.run() outputs "first". Do not change or remove this line.
        synchronized (lock) {
            while (count != 1) {
                lock.wait();
            }
            printFirst.run();
            count = 2;
            lock.notifyAll();
        }
    }

    public void second(Runnable printSecond) throws InterruptedException {

        // printSecond.run() outputs "second". Do not change or remove this line.
        synchronized (lock) {
            while (count != 2) {
                lock.wait();
            }
            printSecond.run();
            count = 3;
            lock.notifyAll();
        }
    }

    public void third(Runnable printThird) throws InterruptedException {

        // printThird.run() outputs "third". Do not change or remove this line.
        synchronized (lock) {
            while (count != 3) {
                lock.wait();
            }
            printThird.run();
            count = 1;
            lock.notifyAll();
        }
    }
}

解法二：Semaphore

class Foo {

    Semaphore second = new Semaphore(0);

    Semaphore third = new Semaphore(0);

    public Foo() {
    }

    public void first(Runnable printFirst) throws InterruptedException {

        // printFirst.run() outputs "first". Do not change or remove this line.
        //
        printFirst.run();
        second.release();
    }

    public void second(Runnable printSecond) throws InterruptedException {

        // printSecond.run() outputs "second". Do not change or remove this line.
        second.acquire();
        printSecond.run();
        third.release();
    }

    public void third(Runnable printThird) throws InterruptedException {

        // printThird.run() outputs "third". Do not change or remove this line.
        third.acquire();
        printThird.run();
    }
}

解法四：CountDownLatch

class Foo {

    CountDownLatch second = new CountDownLatch(1);

    CountDownLatch third = new CountDownLatch(1);

    public Foo() {
    }

    public void first(Runnable printFirst) throws InterruptedException {

        // printFirst.run() outputs "first". Do not change or remove this line.
        //
        printFirst.run();
        second.countDown();
    }

    public void second(Runnable printSecond) throws InterruptedException {

        // printSecond.run() outputs "second". Do not change or remove this line.
        second.await();
        printSecond.run();
        third.countDown();
    }

    public void third(Runnable printThird) throws InterruptedException {

        // printThird.run() outputs "third". Do not change or remove this line.
        third.await();
        printThird.run();
    }
}

解法五：Lock + Conditon

class Foo {

    ReentrantLock lock = new ReentrantLock();

    Condition first = lock.newCondition();

    Condition second = lock.newCondition();

    Condition third = lock.newCondition();

    volatile int count = 1;

    public Foo() {

    }

    public void first(Runnable printFirst) throws InterruptedException {

        // printFirst.run() outputs "first". Do not change or remove this line.
        lock.lock();
        if (count != 1) {
            first.await();
        }
        count = 2;
        printFirst.run();
        second.signal();
        lock.unlock();
    }

    public void second(Runnable printSecond) throws InterruptedException {

        // printSecond.run() outputs "second". Do not change or remove this line.
        lock.lock();
        if (count != 2) {
            second.await();
        }
        count = 3;
        printSecond.run();
        third.signal();
        lock.unlock();
    }

    public void third(Runnable printThird) throws InterruptedException {

        // printThird.run() outputs "third". Do not change or remove this line.
        lock.lock();
        if (count != 3) {
            third.await();
        }
        count = 1;
        printThird.run();
        first.signal();
        lock.unlock();
    }
}

解法六：CyclicBarrier

class Foo {

    CyclicBarrier first = new CyclicBarrier(2);

    CyclicBarrier second = new CyclicBarrier(2);

    public Foo() {

    }

    public void first(Runnable printFirst) throws InterruptedException {

        // printFirst.run() outputs "first". Do not change or remove this line.
        try {
            printFirst.run();
            first.await();
        } catch (BrokenBarrierException e) {
            e.printStackTrace();
        }
    }

    public void second(Runnable printSecond) throws InterruptedException {

        // printSecond.run() outputs "second". Do not change or remove this line.
        try {
            first.await();
            printSecond.run();
            second.await();
        } catch (BrokenBarrierException e) {
            e.printStackTrace();
        }
    }

    public void third(Runnable printThird) throws InterruptedException {

        try {
            second.await();
            printThird.run();
        } catch (BrokenBarrierException e) {
            e.printStackTrace();
        }
    }
}

解法七：LockSupport

class Foo {

    Thread first = null;

    Thread second = null;

    Thread third = null;

    volatile int count = 1;

    public Foo() {

    }

    public void first(Runnable printFirst) throws InterruptedException {

        first = Thread.currentThread();
        // printFirst.run() outputs "first". Do not change or remove this line.
        printFirst.run();
        count = 2;
        LockSupport.unpark(second);
    }

    public void second(Runnable printSecond) throws InterruptedException {

        second = Thread.currentThread();
        // printSecond.run() outputs "second". Do not change or remove this line.
        if (count != 2) {
            LockSupport.park();
        }
        printSecond.run();
        count = 3;
        LockSupport.unpark(third);
    }

    public void third(Runnable printThird) throws InterruptedException {

        third = Thread.currentThread();
        // printThird.run() outputs "third". Do not change or remove this line.
        if (count != 3) {
            LockSupport.park();
        }
        printThird.run();
    }
}
