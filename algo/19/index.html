19. 删除链表的倒数第 N 个结点 - 力扣（LeetCode）

解法

func removeNthFromEnd(head *ListNode, n int) *ListNode {
   num, temp := 0, head
   for temp != nil {
      num++
      temp = temp.Next
   }
   if num-n == 0 {
      return head.Next
   }
   count := num - n - 1
   temp = head
   for count != 0 {
      temp = temp.Next
      count--
   }
   temp.Next = temp.Next.Next
   return head
}
